## ## 作业复盘

```
AIModel（父类）            ← 定义统一接口 predict
   ├── TextModel           ← 重写 predict，模拟文本生成（耗时1秒）
   └── ImageModel          ← 重写 predict，模拟图像识别（耗时2秒）

Scheduler（调度器）        ← 接收多个(用户, 模型, 输入)任务，统一调度
   ├── run_serial()        ← 串行执行所有任务
   └── run_concurrent()    ← 多线程并发执行所有任务

主程序：构建模型和任务 → 分别用两种方式跑 → 打印对比报告
```

```python
import time
import threading
from datetime import datetime

#====1.模型层（aimodel-textmodel/imagemodel)
class AIModel:
    """所有 AI 模型的父类，定义统一接口。"""
    def __init__(self, name, model_type):
        self.name = name
        self.model_type = model_type

    def predict(self, input_data):
        # 父类只定义接口，具体实现交给子类
        raise NotImplementedError("子类必须实现 predict 方法")


class TextModel(AIModel):
    """文本生成模型。"""
    def predict(self, input_data):
        start = datetime.now()
        print(f"  [{self.name}] 正在生成文本：{input_data}")
        time.sleep(1)  # 模拟推理耗时
        end = datetime.now()
        return {
            "output": f"《{input_data}》的生成结果",
            "cost": (end - start).total_seconds()
        }


class ImageModel(AIModel):
    """图像识别模型。"""
    def predict(self, input_data):
        start = datetime.now()
        print(f"  [{self.name}] 正在识别图像：{input_data}")
        time.sleep(2)  # 图像模型更慢
        end = datetime.now()
        return {
            "output": f"{input_data} 识别为：猫",
            "cost": (end - start).total_seconds()
        }


#====2.调度器=========
class Schedule:
    def __init__(self):
        self.records = []
        self.lock = threading.Lock()

    def _run_one(self, user, model, input_data):
        """执行单个用户的请求"""
        start = datetime.now()
        result = model.predict(input_data)
        end = datetime.now()
        total_cost = (end - start).total_seconds()
        #lock.acquire()
        with self.lock:
            self.records.append({
                "user": user,
                "model": model.name,
                "type": model.model_type,
                "input": input_data,
                "output": result["output"],
                "total_cost": total_cost
            })
        #lock.release()

    def run_concurrent(self, tasks):
        """多线程并发执行所有任务。"""
        self.records = []
        threads = []
        for user, model, input_data in tasks:
            t = threading.Thread(
                target=self._run_one,
                args=(user, model, input_data)
            )
            threads.append(t)
            t.start()
        for t in threads:
            t.join()    

    def report(self):
        """打印任务明细。"""
        for r in self.records:
            print(f"  {r['user']} -> {r['model']}({r['type']}) "
                  f"输入:{r['input']} | 输出:{r['output']} | 耗时:{r['total_cost']:.2f}秒")

    # 2. 准备任务：6 个用户同时请求

        ("用户2", text_model,  "讲个冷笑话"),

    ]   

    scheduler = Scheduler()

    # 4. 并发执行

    print("【并发模式】开始执行")
    print("=" * 50)

    scheduler.run_concurrent(tasks)
    c_end = datetime.now()
    c_total = (c_end - c_start).total_seconds()
    scheduler.report()

    print(f"并发总耗时：  {c_total:.2f} 秒")

### 3. 专项全套知识点：`__init__.py` 文件

#### 3.1 区分两个极易混淆概念

1. `__init__.py`：**文件夹内的文件**，标记文件夹为Python包；
2. `def __init__(self):`：**class内部魔法构造方法**，创建对象时自动执行。

#### 3.2 核心问题：`__init__.py` 里面必须写代码吗？

**不需要，空白文件完全生效！**只需要在`goods`文件夹新建一个空白`__init__.py`，无需任何代码，Python即可识别为可导入包。示例目录：
    goods/
    ├── __init__.py  # 文件空白，无代码
    └── base_drink.py

main.py正常导入：
    from goods.base_drink import BaseDrink

#### 3.3 `__init__.py` 两种使用场景

1. **课堂标准写法（仅用于导入）**文件完全空白，简洁通用。

2. **进阶简化导入（可选）**在`__init__.py`预导入类，简化外部代码：
   
   # goods/__init__.py
   
      from goods.base_drink import BaseDrink
      from goods.fruit_tea import FruitTea
   main简写：
      from goods import BaseDrink, FruitTea

#### 3.4 拓展：文件夹没有`__init__.py`如何导入？

   with管理线程锁
   """
   
   import threading
   
   # 【重点补充】全局共享库存字典，所有线程共用，必须提前定义
   global_stock = {"珍珠奶茶": 50, "杨枝甘露": 30}

###3. 类4大方法和__init__构造

####3.1 类的4大方法

* **实例方法**：首参数`self`，操作单个对象独有属性；
* **类方法 @classmethod**：首参数`cls`，操作全类共享类属性；
* **静态方法 @staticmethod**：无self/cls，仅独立工具计算；
* **抽象方法 @abstractmethod**：仅声明无实现，子类必须重写。

#### 3.2 __init__

一种魔法方法，创建实例自动执行，用于初始化`self.xxx`实例属性

class BaseDrink:
    def __init__(self, name):
        self.name = name

class FruitTea(BaseDrink):
    def __init__(self, name):
        super().__init__(name)
        self.type = "果茶"

t = FruitTea("杨枝甘露")
print(t.name) # 正常输出杨枝甘露
```

class FruitTea(BaseDrink):

# 包含：数字校验、价格计算、文件读写、线程锁库存管理等功能

import threading


def check_positive(num: float) -> bool:
    """
    校验数字是否大于0
    :param num: 价格、库存、数量等数字
    :return: 合法大于数返回True，否则返回False
    :raises TypeError: 输入非数字类型时抛出异常
    """
    if not isinstance(num, (int, float)):
        raise TypeError("参数必须传入数字（int/float）")
    return num > 0


def calc_total_price(price: float, num: int) -> float:
    """
    计算饮品总价
    :param price: 单品单价
    :param num: 购买数量
    :return: 应付金额，保留两位小数
    :raises ValueError: 价格或数量小于等于0时抛出异常
    """
    if not check_positive(price) or not check_positive(num):
        raise ValueError("价格、购买数量必须大于0")
    return round(price * num, 2)

def save_order_with_with(order_info: str):
    """
    使用with语法保存订单到order.txt文件
    with上下文管理器会自动关闭文件，杜绝资源泄漏
    :param order_info: 订单信息字符串
    """
    with open("order.txt", "a", encoding="utf-8") as f:
        f.write(order_info + "\n")
    print("订单保存完成，文件已自动关闭")


def read_all_orders():

    使用with读取全部历史订单
    :return: 订单文件全部内容字符串
    """
    with open("order.txt", "r", encoding="utf-8") as f:

def clear_order_file():
    """使用with w模式清空订单文件"""
    with open("order.txt", "w", encoding="utf-8") as f:
        pass
    print("订单文件已清空")


# ==================== 列表推导式工具 ====================

def get_cheap_drinks(drink_dict: dict, limit: float) -> list:
    """
    使用列表推导式筛选低于指定价格的饮品
    :param drink_dict: 饮品名称-价格字典
    :param limit: 价格上限
    :return: 低于限价的饮品名称列表
    """
    if not isinstance(drink_dict, dict):
        raise TypeError("参数必须是饮品名称-价格字典")
    cheap_data = {name: p for name, p in drink_dict.items() if p < limit}
    return list(cheap_data.keys())


# ==================== 生成器工具 ====================

def order_record_generator(order_list: list):
    """
    订单生成器：按需生成订单记录，节省内存
    :param order_list: 订单列表，每项为(饮品名, 数量, 总价)元组
    :yield: 格式化订单记录字符串
    """
    for order in order_list:
        yield f"饮品：{order[0]}，数量：{order[1]}，总价：{order[2]}"


# ==================== 多线程库存管理 ====================

# 全局共享库存字典（所有线程共用，必须提前定义）
global_stock = {"珍珠奶茶": 50, "杨枝甘露": 30, "芝士葡萄": 40, "美式咖啡": 60}
# 创建互斥锁对象，保证多线程安全
stock_lock = threading.Lock()


def sell_drink_thread_safe(drink_name: str, sell_num: int):
    """
    使用with线程锁安全扣减库存，防止多线程超卖
    with lock 会自动执行 acquire() 和 release()，避免死锁
    :param drink_name: 饮品名称
    :param sell_num: 售卖数量
    :return: 售卖成功返回True，失败返回False
    """
    with stock_lock:
        if global_stock[drink_name] < sell_num:
            print(f"{drink_name}库存不足，售卖失败")
            return False
        global_stock[drink_name] -= sell_num
        info = f"{drink_name} 售出{sell_num}杯，剩余库存{global_stock[drink_name]}"
        save_order_with_with(info)
        print(info)
        return True


def multi_thread_sell():
    """多线程并发售卖测试函数"""
    t1 = threading.Thread(target=sell_drink_thread_safe, args=("珍珠奶茶", 5))
    t2 = threading.Thread(target=sell_drink_thread_safe, args=("珍珠奶茶", 3))
    t1.start()
    t2.start()
    t1.join()
    t2.join()

sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from abc import ABC, abstractmethod
from shop_tools import check_positive


class BaseDrink(ABC):
    """
    饮品抽象基类：所有具体饮品必须继承此类
    ABC = Abstract Base Class，抽象类无法直接实例化
    子类必须重写 @abstractmethod 装饰的抽象方法
    """

    # 类属性：门店全场折扣（所有饮品共享）
    shop_discount = 1.0

    def __init__(self, name: str, price: float):
        """
        构造方法：创建饮品实例时自动执行，初始化实例属性
        :param name: 饮品名称
        :param price: 饮品单价
        """
        self.name = name
        self.price = price
        # 私有库存属性（双下划线开头，外部不可直接访问）
        self.__stock = 50

    # ==================== 1. 实例方法 ====================
    # 操作单个对象独有属性，首参数为 self

    def get_stock(self) -> int:
        """获取当前饮品库存"""
        return self.__stock

    def sell(self, num: int):
        """
        售卖扣减库存
        :param num: 售卖数量
        :raises ValueError: 库存不足或数量非法时抛出异常
        """
        if not check_positive(num) or num > self.__stock:
            raise ValueError("库存不足或数量非法")
        self.__stock -= num
        print(f"售出{num}杯{self.name}，剩余库存：{self.__stock}")

    # ==================== 2. 类方法 ====================
    # 操作全类共享的类属性，首参数为 cls，使用 @classmethod 装饰

    @classmethod
    def set_shop_discount(cls, discount: float):
        """
        设置门店全场折扣
        :param discount: 折扣值（0~1之间，如0.9表示9折）
        :raises ValueError: 折扣范围不合法时抛出异常
        """
        if 0 < discount <= 1:
            cls.shop_discount = discount
            print(f"门店折扣更新为：{discount}")
        else:
            raise ValueError("折扣范围0~1")

    # ==================== 3. 静态方法 ====================
    # 独立工具方法，无需self/cls，使用 @staticmethod 装饰

    @staticmethod
    def check_drink_name(name: str) -> bool:
        """校验饮品名称是否合法（非空字符串）"""
        return len(name.strip()) > 0

    # ==================== 4. 抽象方法 ====================
    # 仅声明无实现，子类必须重写，否则实例化时报错

    @abstractmethod
    def get_final_price(self, buy_num: int) -> float:
        """
        抽象计价方法：计算购买指定数量的最终价格
        子类必须根据各自优惠规则重写此方法
        :param buy_num: 购买数量
        :return: 最终应付金额
        """
        pass

    def print_ticket(self, buy_num: int):
        """
        通用打印小票方法（可被子类重写实现差异化小票）
        :param buy_num: 购买数量
        """
        total = self.get_final_price(buy_num)
        print(f"====={self.name}消费小票=====")
        print(f"购买{buy_num}杯，应付总价：{total}元")


# ==================== 模块测试代码 ====================
# 注意：BaseDrink 是抽象基类，无法直接实例化测试
# 此处仅测试静态方法和类方法，实例方法需通过子类测试
if __name__ == "__main__":
    print("===== base_drink.py 模块自测 =====\n")

    # 1. 测试静态方法：名称校验

class Coffee(BaseDrink):
    """
    咖啡类：继承饮品基类
    优惠规则：全场折扣基础上统一88折
    """

    def __init__(self, name: str, price: float):
        """
        咖啡构造方法
        :param name: 咖啡名称
        :param price: 咖啡单价
        """
        super().__init__(name, price)  # 调用父类构造

    def get_final_price(self, buy_num: int) -> float:
        """
        重写抽象计价方法：咖啡专属优惠
        计算公式：原价 × 全场折扣 × 0.88
        :param buy_num: 购买数量
        :return: 优惠后总价
        """
        origin = self.price * buy_num
        final = origin * self.shop_discount * 0.88
        return round(final, 2)
